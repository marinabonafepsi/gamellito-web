================================================================================
                        FLUXO DE COMPARTILHAMENTO
================================================================================

FAMILIA
  │
  ├─ Cria criança (Marina)
  │
  ├─ Faz registros (glicemia, alimentação)
  │
  └─ Compartilha com Profissional
      │
      ├─ POST /familia/crianca/{id}/compartilhar
      │  └─ Tipo de acesso: readonly | comment | full
      │
      ├─ Sistema cria:
      │  ├─ permissao (usuario_dono → usuario_acesso)
      │  ├─ convite (envia email com token)
      │  └─ consentimento_granular (rastreia consentimento)
      │
      └─ Envia convite por email
         │
         PROFISSIONAL recebe email
         │
         ├─ Se já tem conta:
         │  └─ POST /convites/{id}/aceitar
         │
         ├─ Se é novo usuário:
         │  └─ POST /convites/{token}/aceitar-anonimo
         │     (cria conta + aceita convite)
         │
         └─ Agora pode acessar:
            ├─ GET /profissional/dashboard (vê Marina)
            ├─ GET /profissional/paciente/{id} (detalhes)
            ├─ GET /profissional/paciente/{id}/registros
            └─ POST /profissional/paciente/{id}/notas (comentários)

================================================================================
                      SEQUÊNCIA DE MULTI-TENANT
================================================================================

Institução (Escola)
    │
    ├─ Cria Admin → usuario com role='instituicao'
    │
    ├─ Admin cria Grupos:
    │  ├─ Turma 5A (educador_id)
    │  ├─ Turma 6B (educador_id)
    │  └─ Turma 7C (educador_id)
    │
    ├─ Educador adiciona alunos ao grupo:
    │  └─ grupos_membros (usuario_id, grupo_id)
    │
    └─ Alunos (Familia com crianca):
       ├─ Fazem registros
       └─ Educador pode ver:
          ├─ GET /educador/grupo/{id}
          ├─ GET /educador/grupo/{id}/relatorios
          └─ POST /instituicao/relatorios/exportar


================================================================================
                        TABELAS PRINCIPAIS (10)
================================================================================

1. profissionais
   id, usuario_id (FK unique), crm, especialidade, instituicao_id, verificado

2. instituicoes
   id, nome, tipo, cnpj (unique), email, admin_id (FK), ativa

3. permissoes
   id, usuario_dono_id (FK), usuario_acesso_id (FK), tipo_acesso,
   expira_em, revogada_em, UNIQUE(usuario_dono_id, usuario_acesso_id)

4. grupos
   id, instituicao_id (FK), nome, descricao, educador_id (FK),
   UNIQUE(instituicao_id, nome)

5. grupos_membros
   id, grupo_id (FK), usuario_id (FK), rol, removido_em,
   UNIQUE(grupo_id, usuario_id)

6. consentimento_granular
   id, usuario_crianca_id (FK), usuario_profissional_id (FK),
   tipo_consentimento, consentimento_dado, ip_origem, user_agent

7. convites
   id, usuario_remetente_id (FK), email_destinatario,
   tipo_convite, token_convite (unique), aceito, expira_em

8. loja_items
   id, nome, tipo, custo_moedas (>0), imagem_url, ativo

9. inventario_usuario
   id, usuario_id (FK), item_loja_id (FK), quantidade,
   ativo, UNIQUE(usuario_id, item_loja_id)

10. (auth.users alterada)
    id, email (unique), role, tenant_id (FK), email_confirmed, deleted_at
